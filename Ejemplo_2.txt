using System;

namespace Ejemplo2
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Ingrese un n√∫mero: ");
            int numero = int.Parse(Console.ReadLine());

            Console.Write("Ingrese el exponente: ");
            int exponente = int.Parse(Console.ReadLine());

            int resultado = Potencia(numero, exponente);

            Console.WriteLine($"{numero}^{exponente} = {resultado}");
        }

        
        static int Potencia(int baseNum, int exp)
        {
            int resultado = 1;

            for (int i = 1; i <= exp; i++)
            {
                resultado *= baseNum;
            }

            return resultado;
        }
    }
}
