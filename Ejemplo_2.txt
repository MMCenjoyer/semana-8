using System;

namespace Ejemplo2
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Ingrese un número: ");
            int numero = int.Parse(Console.ReadLine());

            Console.Write("Ingrese el exponente: ");
            int exponente = int.Parse(Console.ReadLine());

            // Llamamos al método de la clase EJERCICIO2
            int resultado = EJERCICIO2.Potencia(numero, exponente);

            Console.WriteLine($"{numero}^{exponente} = {resultado}");
        }
    }
}
namespace Ejemplo2
{
    public class EJERCICIO2
    {
        public static int Potencia(int baseNum, int exp)
        {
            int resultado = 1;

            for (int i = 1; i <= exp; i++)
            {
                resultado *= baseNum;
            }

            return resultado;
        }
    }
}
